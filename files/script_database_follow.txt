CREATE TABLE motorista_parceiros (
    id SERIAL PRIMARY KEY,

    -- Identificação
    nome_completo VARCHAR(255),
    data_nascimento DATE,
    documento_numero VARCHAR(100),
    documento_tipo VARCHAR(50),
    rg VARCHAR(50),
    uf_identificacao VARCHAR(2),
    orgao_emissor VARCHAR(100),
    estado_civil VARCHAR(50),

    -- Contato
    telefone VARCHAR(20),
    email VARCHAR(255),

    -- Contatos referenciais (JSON array de objetos)
    contatos_referenciais JSON,

    -- Endereço
    estado VARCHAR(2),
    cidade VARCHAR(100),
    bairro VARCHAR(100),
    rua VARCHAR(255),
    numero VARCHAR(20),

    -- Documentação profissional (CNH)
    cnh_numero VARCHAR(50),
    cnh_categoria VARCHAR(10),
    cnh_validade DATE,
    cnh_uf_emissor VARCHAR(2),
    foto_cnh JSON,

    -- Dados do veículo
    placa VARCHAR(20),
    marca VARCHAR(100),
    renavam VARCHAR(50),
    modelo VARCHAR(100),
    crv VARCHAR(50),
    ano INT,
    tipo VARCHAR(50),
    cor VARCHAR(50),
    capacidade_carga DECIMAL(10, 2),
    medida_de_capacidade VARCHAR(20),
    foto_veiculo JSON
);
